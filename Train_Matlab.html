function lgraph = buildMobilenetv2(numClasses, inputSize, classNames, classWeights)
lgraph = layerGraph(mobilenetv2);
% Modify to allow gray image
lgraph = lgraph.replaceLayer('input_1', imageInputLayer([inputSize inputSize 1],...
'Normalization', 'zscore', 'Name','input_1'));
lgraph = lgraph.replaceLayer('Conv1', convolution2dLayer(3, 32, 'Stride', 2,...
'Padding', 'same', 'Name', 'Conv1'));
[learnableLayer,~] = findLayersToReplace(lgraph);
newLearnableLayer = fullyConnectedLayer(numClasses, 'Name','new_fc');
lgraph = replaceLayer(lgraph, learnableLayer.Name, newLearnableLayer);
lgraph = replaceLayer(lgraph, 'ClassificationLayer_Logits',...
classificationLayer('Name', 'Output', 'Classes',classNames, 'ClassWeights',classWeights));
end
%--------------------------------------------------------------------------
function options = setupOptions(optimizer, miniBatchSize, val, valFrequency, checkpointPath)
if ~exist(checkpointPath, 'dir')
mkdir(checkpointPath)
end
options = trainingOptions(...
optimizer,...
'MiniBatchSize' , miniBatchSize,...
'MaxEpochs' , 30,...
'shuffle' , 'every-epoch',...
'InitialLearnRate' , 3e-3,...
'LearnRateSchedule' , 'piecewise',...
'ValidationData' , val,...
'ValidationFrequency' , valFrequency,...
'ValidationPatience' , 5,...
'ExecutionEnvironment', 'multi-gpu',...
'Checkpoint' , checkpointPath);
end
